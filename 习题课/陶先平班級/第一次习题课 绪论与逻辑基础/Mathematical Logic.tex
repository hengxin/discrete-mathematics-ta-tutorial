\subsection{Propositional Logic}

\frame
{
  \frametitle{Propositional Logic---Proposition(命题)}

  \begin{quote}
    ``命题'' 主要是指一些字或者其它符号组合成的一种形式，这种形式所表达的或者为真或者为假。---罗素
  \end{quote}

  \vspace{0.2cm}
  \color{gray}{Notice: 排中律。有的逻辑学家并不承认这一规律。}
   \vspace{0.3cm}
  \pause
  \begin{itemize}
    \item 原子命题:  $p,q,r,s\dots$ ($\because$我们不太关心具体的命题内容)
    \vspace{0.25cm}
    \item 复合命题:  $\neg, \lor, \land, \to, \leftrightarrow$
    \vspace{0.25cm}
    \item 命题公式及其解释(interpretation)
    \vspace{0.25cm}
    \item 重言式(tautology), 矛盾式(contradiction)
  \end{itemize}
}

\frame
{
  \frametitle{Propositional Logic---命题逻辑等值演算}

  Question: 如何判断两个命题公式是等值的？
  \begin{enumerate}
    \item 真值表
    \vspace{0.20cm}
    \item 等值式
  \end{enumerate}

  \vspace{0.50cm}

  \pause
  重要等值式：
  \begin{description}
    \item [德摩根律 De Morgan]
      \[ \neg(A \lor B) \Leftrightarrow \neg A \land \neg B \]
      \[ \neg(A \land B) \Leftrightarrow  \neg A \lor \neg B \]
    \item [蕴涵等值式]          $A \to B \Leftrightarrow \neg A \lor B$
    \item [假言易位]            $A \to B \Leftrightarrow \neg B \to \neg A$
    \item [归谬论]              $(A \to B) \land (A \to \neg B) \Leftrightarrow \neg A$
  \end{description}
}


\frame
{
  \frametitle{Propositional Logic---推理的形式结构}
  \textbf{推理的形式结构}
  \[ \lbrace A_1,A_2,\cdots,A_n \rbrace \vdash B \]
  \vspace{0.50cm}
  \textbf{\alert{推理正确}当且仅当}
  \[ A_1 \land A_2 \land \cdots A_n \to B \]
  为重言式。(等值演算)
}

\frame
{
  \frametitle{Propositional Logic---自然推理系统}
  \textbf{Q: 如何保证推理的正确性?} \\
  \vspace{0.3cm}
  \textcolor{red}{A: 推理规则集$R$}

  \vspace{0.5cm}
  \begin{quote}
    ``\textcolor{blue}{证明}是一个描述推理过程的命题公式序列，其中的每个公式或者是已知前提，或者是由前面的
    公式应用\textcolor{blue}{推理规则}得到的结论(中间结论或推理中的结论)。
  \end{quote}
}

\subsection{Predicate Logic}

\frame
{
  \frametitle{Predicate Logic---符号化}
  \begin{enumerate}
    \item 个体词
      \begin{enumerate}
        \item 个体常项
        \item 个体变项
      \end{enumerate}
    \vspace{0.25cm}
    \item 谓词(predicate) \\
      $P(x_1,x_2,\cdots,x_n) \to \lbrace \top , \bot \rbrace$
    \vspace{0.25cm}
    \item 量词
      \begin{enumerate}
        \item 全称量词 : $\forall x P(x)$
        \item 存在量词 : $\exists x P(x)$
      \end{enumerate}
  \end{enumerate}
}

\frame
{
  \frametitle{Predicate Logic---等值演算}

  \textcolor{gray}{只要求了解，会进行简单推理即可}
  \vspace{0.30cm}
  \begin{enumerate}
  \only<1>
  {
    \item $\forall x P(x) \vdash P(c) \text{ if } c \in U$
    \vspace{0.20cm}
    \item $\lbrace P(c) \text{ 对任意 } c \in U \rbrace \vdash \forall x P(x)$
    \vspace{0.20cm}
    \item $\exists x P(x) \vdash P(c) \text{ 对某个 } c \in U$
    \vspace{0.20cm}
    \item $\lbrace \text{对某个 } c \in U, P(c) \rbrace \vdash \exists x P(x)$
  }
  \only<2>
  {
    \vspace{0.20cm}
    \item $\neg \forall x A(x) \Leftrightarrow \exists x \neg A(x)$
    \vspace{0.20cm}
    \item $\neg \exists x A(x) \Leftrightarrow \forall x \neg A(x)$
  }
  \end{enumerate}
}

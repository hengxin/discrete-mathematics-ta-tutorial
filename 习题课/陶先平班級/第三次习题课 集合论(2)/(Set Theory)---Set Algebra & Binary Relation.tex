  \begin{center}
    闭包是一个很重要也很常见的概念。
  \end{center}

  \begin{columns}
    \column{0.5\textwidth}
      \begin{description}
        \item[集合] 有序对集合，即二元关系
        \item[性质] 传递性
        \item[闭包] 添加有序对，扩充集合使其满足传递性
      \end{description}
    \column{0.5\textwidth}
      \begin{description}
        \item[集合] 整数
        \item[运算] 除法
        \item[闭包] 扩充为有理数，使其对于除法操作封闭
      \end{description}
  \end{columns}
}

\frame
{
  \frametitle{Closure}

  \begin{columns}[c]
    \column{0.6\textwidth}
      $R$为关系，$R$的自反(对称或传递)闭包是关系$R'$ \footnote{The way to prove...}:
      \vspace{0.20cm}
      \begin{enumerate}
        \setlength{\itemsep}{0.20cm}
        \item $R'$是自反的(对称或传递的)
        \item $R \subseteq R'.$
        \item \textcolor{blue}{$R'$具有最小性}
      \end{enumerate}
    \column{0.4\textwidth}
      \begin{enumerate}
        \setlength{\itemsep}{0.20cm}
        \item $r(R) = R \cup R^{0}$
        \item $s(R) = R \cup R^{-1}$
        \item $t(R) = R \cup R^{2} \cup R^{3} \cup \cdots$
      \end{enumerate}
  \end{columns}
}

\subsubsection{Partial Order Set}

\frame
{
  \frametitle{Partial Order Relation}
  如果集合$A$上的非空关系$R$具有自反性，反对称性和传递性，则称$R$为\textit{偏序关系}，记作 $\preceq$。
  $\langle A,\preceq \rangle$为\textit{偏序集}(poset)。
  \vspace{0.40cm}

  偏序集的表示：哈斯图(Hasse Diagram).
  \vspace{0.40cm}

  偏序集中的特殊元素：
  \begin{enumerate}
    \setlength{\itemsep}{0.20cm}
    \item 最大元，最小元，极大元，极小元。
    \item 上界，下界，最小上界，最大下界。
    \item 有穷集合一定有极大元和极小元且不一定唯一。某元素可能既是极大元又是极小元。
    \item 最大元和最小元不一定存在。若存在则必唯一。若存在，则最元同时也是极元。
  \end{enumerate}

}

\frame
{
  \frametitle{Function}

  函数是一种特殊的二元关系。  \\
  \vspace{0.20cm}

  定义：二元关系$F$可被称为函数，如果$\forall x \in dom(F), \exists! y \in ran(F) \land xFy.$  \\
  \vspace{0.40cm}

  函数也是集合: $Function \to \textrm{Binary Relation} \to \textrm{Ordered Pair} \to Set.$  \\
  \vspace{0.40cm}
}

\frame
{
  \frametitle{Function}

  \textbf{函数的性质: } $f : A \to B,$
  \begin{description}
    \setlength{\itemsep}{0.20cm}
    \item[单射(Injective)] $(\forall x_1,x_2 \in A)(f(x_1) = f(x_2) \leftrightarrow x_1 = x_2)$
    \item[满射(Surjective)] $(\forall y \in B)(\exists x \in A, f(x) = y)$
    \item[双射(Bijective)] $\textrm{Injective } + \textrm{Surjective}$
  \end{description}

  \vspace{0.30cm}
  \textbf{函数的复合:}
  设$F,G,H$为函数，则
  \begin{itemize}
    \item $(F \circ G) \circ H)$ 与 $F \circ (G \circ H)$都是函数。
    \item $(F \circ G) \circ H) = F \circ (G \circ H)$
  \end{itemize}

  \vspace{0.30cm}
  \textbf{反函数:} \\
  \vspace{0.15cm}
  对于双射函数$f: A \to B$, $f^{-1}: B \to A$是它的反函数。
}

\frame
{
  \frametitle{Cardinality of Set}

  \textbf{有限集合与大数}
    \begin{enumerate}
      \setlength{\itemsep}{0.20cm}
      \item 据不少非洲探险家证实，在某些原始部落里，不存在比\textbf{3}大的数词。
      \item 国际象棋。$2^{64} - 1 = \textrm{全世界在2000年内所生产的全部小麦}.$
      \item 汉诺塔问题。$2^{64} - 1 = \textrm{5800亿年}.$
      \item 莎士比亚作品自动印刷机。
    \end{enumerate}

  \vspace{0.30cm}
  \alert{如何比较无穷集合的大小?} \\
  \vspace{0.15cm}
  \textbf{集合等势:}  \\
  $A \approx B \Leftrightarrow \exists f: A \to B, f \textrm{ is bijective}.$  \\
  \vspace{0.20cm}
  \textbf{集合优势:}  \\
  $A \preceq \cdot B \Leftrightarrow \exists f: A \to B, f \textrm{ is injective}.$  \\
  \vspace{0.20cm}
  $A \preceq \cdot B \land B \preceq \cdot A \Rightarrow A \approx B.$
}

  \textbf{同态(Homomorphism):}  \\
  \vspace{0.20cm}
  设$V_1 = \langle A, \circ \rangle$ 和 $V_2 = \langle B,\ast \rangle$是同类型的代数系统，  \\

  \vspace{0.30cm}

  \end{quote}

  \begin{itemize}

    \item 单同态

    \item 自同态，单自同态，满自同态，自同构
  \end{itemize}

\frame
{
  \frametitle{Closure}

    \textbf{复合闭包:}  \\
    \vspace{0.20cm}
    证明: $A \textrm{ is finite set,} R \subseteq A \times A \Rightarrow rt(R) = tr(R).$  \\
    \vspace{0.20cm}
    \textbf{提示:}  \\
    $(I_A \cup R)^n = I_A \cup R \cup R^2 \cup \cdots \cup R^n = I_A \cup (\cup_{i=1}^{n}R^i)$  \\
    \vspace{0.20cm}
    \textbf{解答:}  \\
    \pause
  \normalsize
    \begin{equation*}
        \begin{array}{rcl}
            tr(R) & = & t(I_A \cup R)  \\
                  & = & \cup_{i=1}^{n}(I_A \cup R)^{i}  \\
                  & = & (I_A \cup R) \cup (I_A \cup R)^2 \cup \cdots \cup (I_A \cup R)^n  \\
                  & = & (I_A \cup R) \cup (I_A \cup (\cup_{i=1}^{2}R^i)) \cup \cdots \cup (I_A \cup (\cup_{i=1}^{n}R^i))  \\
                  & = & I_A \cup R \cup R^2 \cup \cdots \cup R^n  \\
                  & = & I_A \cup (\cup_{i=1}^{n}R^i)  \\
                  & = & I_A \cup t(R)  \\
                  & = & r(t(R))  \\
                  & = & rt(R)
        \end{array}
    \end{equation*}
}

\frame
{
  \frametitle{Closure}

    \textbf{复合闭包:}  \\
    \vspace{0.20cm}
    \textbf{证明}: $R \subseteq A \times A \Rightarrow st(R) \subseteq ts(R).$  \\
    \vspace{0.40cm}
    \textbf{解答:}  \\

%  \normalsize
    \begin{equation*}
        \begin{array}{rcl}
            R\subseteq s(R)
                  & \Rightarrow & t(R) \subseteq t(s(R))  \\
                  & \Rightarrow & st(R) \subseteq \textcolor{red}{sts(R) = ts(R)}
        \end{array}
    \end{equation*}

    \vspace{0.30cm}
    $s(R)[sym] \Rightarrow ts(R)[sym] \Rightarrow sts(R) = ts(R).$
}

\begin{frame}{Permutation Group}

	    \begin{itemize}
	    \setlength{\itemsep}{0.30cm}
	      \item \emph{对称群(Symmetric group)}:   \\
	        非空集合$X$上的一一变换关于合成运算所构成的群$S_X$.
	      \item \emph{变换群(Transformation group)}: $S_X$的任一子群.

	         $X$上的一一变换
	      \item \emph{$n$次对称群:}  $S_n$
	      \item \emph{置换群(Permutation group):} $S_n$的子群.
	    \end{itemize}
	\end{columns}

\begin{frame}{Cayley Theorem}
  \textbf{(Cayley Theorem, 1854)}  \\
  每一个群都同构于一个变换群.  \\
  \vspace{0.20cm}
  每一个有限群都同构与一个置换群.

  \vspace{0.30cm}
  \textbf{证明要点:}  \\
  \begin{enumerate}
    \item $a \in G$, $\phi_a: \phi_a(x) = ax, \forall x \in G.$ \textcolor{blue}{$\phi_a$ is a function.}
    \item $G_l = \lbrace \phi_a \mid a \in G. \rbrace$ 关于变换的合成构成$S_G$的子群.
    \item $\rho: G \cong G_l. a \mapsto \phi_a, \forall a \in G.$
  \end{enumerate}

  \pause
  \vspace{0.30cm}
  \textbf{重要意义:}  \\
  \scriptsize
  \begin{quote}
    ``Cayley's theorem puts all groups on the same footing, by considering any group
    (including infinite groups such as $(\mathbf{R},+)$) as a permutation group of some underlying set.  \\
    Thus, theorems which are true for permutation groups are true for groups in general.''
  \end{quote}
\end{frame}

\begin{frame}{Permutation $\to$ Transposition}
  \textbf{对换:}  \\
  每一置换都可表为对换的乘积.
  \[
    \sigma = (i_1 i_2 \cdots i_r) = (i_1 i_2)(i_1 i_3) \cdots (i_1 i_r).

\begin{frame}{Permutation $\to$ Transposition}
  \textbf{Parity of Permutation}  \\
  每一个置换表为对换的乘积，所用对换个数的奇偶性是唯一的.  \\

  \vspace{0.20cm}
  \textbf{证明要点:}  \\
  \textcolor{blue}{Insight:} $\sigma$的对换表示中对换个数与排列$i_1,i_2,\cdots,i_n$的逆序数同奇偶(唯一).

  \vspace{0.20cm}
  群同态:
  \[
    \textrm{sgn}: S_n \to \lbrace -1,1 \rbrace.
  \]

\end{frame}

\begin{frame}{Permutation $\to$ Transposition}
  \textbf{计数:}  \\
  $n \geq 2$阶置换中，偶置换与奇置换各有$\frac{n!}{2}$个. \\
  \vspace{0.20cm}
  \begin{itemize}
    \item $f: A_n \to B_n: p \in A_n, f(p) = (a_1,a_2)p.$
    \item $g: B_n \to A_n: p \in B_n, g(p) = (a_1,a_2)p.$
  \end{itemize}

  \vspace{0.20cm}
  \textbf{$n$次交错群(Alternating group)}  \\
  在$S_n$中，全体偶置换构成$S_n$的子群,称为$n$次交错群.

  \vspace{0.20cm}
  \textbf{举例:}
  \[
    A_3 = \lbrace (1),(1 2 3),(1 3 2) \rbrace.
  \]

\subsection{Normal Subgroup and Quotient Group}
\begin{frame}{Normal Subgroup}
\textbf{正规子群:}  \\
$(H, \ast) \triangleleft (G,\ast) : (H,\ast) \leq (G,\ast), (\forall a \in G)(aH = Ha).$  \\
\vspace{0.20cm}
\textcolor{blue}{特别提醒:$\forall h \in H, \exists h' \in H, gh = h'g(g \in G).$}

\vspace{0.30cm}
\textcolor{gray}{
\textbf{举例:}  \\
\begin{itemize}
  \item 平凡群$(\lbrace e \rbrace, \ast)$和$G$是正规子群。($(\forall a \in G)(aG = G = Ga).$)
  \item 交换群$G$的一切子群都是$G$的正规子群。
  \item 群$G$的中心$C$是$G$的正规子群。 \\
    ($C = \lbrace a \mid a \in G \land \forall x \in G (ax = xa) \rbrace.$)
\end{itemize}}
\end{frame}


\begin{frame}{Normal Subgroup}
\textbf{正规子群判定定理:}  \\
$G$为群，$H$为$G$的子群，则:  \\
\[
  H \triangleleft G \Leftrightarrow (\forall g \in G, \forall h \in H)(ghg^{-1} \in H).
\]

\end{frame}


\begin{frame}{Quotient Group}
  \textbf{商群:}  \\
  设$G$是群，$H$是$G$的正规子群，则$H$的所有陪集组成的集合
  \[
    G/H = \lbrace Ha \mid a \in G \rbrace.
  \]
  关于陪集的乘法运算$(Ha) \cdot (Hb) = H(ab)$构成群，称为$G$关于$H$的商群。

  \vspace{0.20cm}
  \textcolor{blue}{正规子群的陪集集合提供了看待群的一种视角。一种视角就是一种抽象。它与普通子群的抽象的区别在于它
  保持了运算。}  \\
\scriptsize
  \pause
  \textbf{证明:}  \\
  \begin{enumerate}
    \item 运算的封闭性 ($Ha \cdot Hb = H(ab) \in G/H$.)
    \item 结合律 ($(Ha \cdot Hb) \cdot Hc = Ha \cdot (Hb \cdot Hc)$.)
    \item 单位元 ($He = H$.)
    \item 逆元   ($(Ha)^{-1} = Ha^{-1}$.)
  \end{enumerate}

\normalsize
  \pause
  \textcolor{blue}{Q: 此证明哪里体现了正规子群的必要性?  \\
  如果没有，为什么只考虑在正规子群之上定义商群呢?}
\end{frame}


\begin{frame}{Quotient Group}
  \textbf{良定义(well-defined)的运算:}  \\
  \[
    (Ha) \cdot (Hb) = H(ab).
  \]
  \textcolor{blue}{要求: \emph{$H$的任意两个陪集$Ha,Hb$的乘积是唯一确定的,与陪集代表元的选取无关}}  \\
  $Ha' = Ha, Hb' = Hb$,  \\
  $Ha' \cdot Hb' = H(a'b') = (Ha')b' = (Ha)b' = (aH)b'$ \\
  $= a(Hb') = a(Hb) = (aH)b = (Ha)b = H(ab) = (Ha) \cdot (Hb).$

  \pause
  \vspace{0.20cm}
  \textcolor{gray}{举例:}  \\
  在群$S_3$中，$H = \lbrace (1),(1,2) \rbrace$.  \\
  $H(1,3) = \lbrace (1,3),(1,3,2) \rbrace = H(1,3,2).$  \\
  $H(2,3) = \lbrace (2,3),(1,2,3) \rbrace = H(1,2,3).$  \\
  然而,
  $H(1,3) \cdot H(2,3) = H(1,2,3).$  \\
  $H(1,3,2) \cdot H(1,2,3) = H(1).$
\end{frame}

\begin{frame}{Fundamental Thorem over Homomorphism}
  \textbf{群同态基本定理:}  \\
  设$\sigma$是群$G$到群$G'$的满同态，$K = Ker \sigma$, 则
  \[
    G/K \cong G'.
  \]

  \vspace{0.20cm}
  \begin{itemize}
    \item 从同构的观点看，群的同态像就是群的商群.
    \item 同态核可以看作是群与其同态像之间相似度的一个度量.
  \end{itemize}
\end{frame}

\begin{frame}{Fundamental Thorem over Homomorphism}
  \textbf{利用群同态基本定理证明商群同构:}  \\
  (1) \emph{第一同构定理:}  \\
	设$\phi$是群$G$到$G'$的同态.则
	\[
	  G / \textrm{Ker} \phi \cong \phi(G).
	\]
  (2) \emph{第二同构定理:}  \\
    设$H \leq G, K \triangleleft G$, 则
    \[
      HK/K \cong H/(H \cap K).
    \]
  (3) \emph{第三同构定理:}  \\
    设$H \triangleleft G, K \triangleleft G, K \subseteq H$，则
    \[
      G/H \cong (G/K)/(H/K).
    \]
\end{frame}

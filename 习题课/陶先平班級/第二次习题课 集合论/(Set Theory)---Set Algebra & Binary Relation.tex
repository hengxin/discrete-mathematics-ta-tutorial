  \frametitle{集合的运算 --- $\cup(union),\cap(intersection),\sim(complement)$}

  \textbf{集合运算律: \scriptsize (compare $P_{93}$ with $P_{18}$)}\\
  \normalsize
  \begin{description}
    \item [Distributive Law]
      \[
        A \cup (B \cap C) = (A \cup B) \cap (A \cup C)
      \]
      \[
        A \cap (B \cup C) = (A \cap B) \cup (A \cap C)
      \]
    \item[De Morgan Law]
      \[
        A - (B \cup C) = (A - B) \cap (A - C)
      \]
      \[
        A - (B \cap C) = (A - B) \cup (A - C)
      \]
    \item[Complement]
      \[
        A - B = A \cap \sim B = A - (A \cap B)
      \]
      \[
        A \cap (B - A) = {\O}
      \]
      \[
        A \cup (B - A) = A \cap B
      \]
  \end{description}
}

\frame
{
  \frametitle{集合的运算 --- 幂集(Power Set)}

  \textbf{定义:}
  \[
    P(A) = \lbrace x \mid x \subseteq A \rbrace
  \]

  \vspace{0.25cm}
  \textbf{Theorem}  \\
    \[
      \mid A \mid = n \in N \to \mid P(A) \mid = 2 ^ n
    \]

  \vspace{0.20cm}
  \textbf{Proof}  \\
    Relation to binomial theorem.
    \[
      {n \choose 0} + {n \choose 1} + \cdots + {n \choose k} + \cdots + {n \choose n} = 2 ^ n \qed
    \]

  \vspace{0.20cm}
  \alert{\textbf{Q: What happens if $A$ is an infinite set?}} \\

  \frametitle{二元关系及其运算}

  \textbf{定义:}  \\
  \vspace{0.20cm}
    $\textrm{若 }R \subseteq A \times B ,\textrm{则称}R\textrm{为从}A\textrm{到}B\textrm{的二元关系}.$

  \vspace{0.30cm}
  \textbf{关系的运算}  \\
  \begin{enumerate}
    \item \textbf{Inverse:} $(F \circ G)^{-1} = G^{-1} \circ F^{-1}$
    \vspace{0.30cm}
    \item \textbf{Composition:}
      \begin{enumerate}
        \item \textbf{Associative Law:} $(F \circ G) \circ H = F \circ (G \circ H)$
        \vspace{0.20cm}
        \item \textbf{Distributive Law over Union:} $F \circ (G \cup H) = F \circ G \cup F \circ H$
        \vspace{0.20cm}
        \item {\textcolor{gray}{Distributive Law over Intersection}:} $F \circ (G \cap H) \subseteq F \circ G \cap F \circ H$ \scriptsize (课本$P_{109}$定理7.4,$P_{132}$第18题)
      \end{enumerate}
  \end{enumerate}
}


\frame
{
  \frametitle{二元关系的性质}

  \textbf{二元关系的重要性质:}
  \vspace{0.20cm}
  \begin{description}
    \item [reflexive:] $(\forall x \in A) (xRx)$
    \vspace{0.20cm}
    \item [irreflexive:] $(\forall x \in A) (\neg xRx)$
    \vspace{0.20cm}
    \item [symmetric:] $(\forall x,y \in A) (xRy \to yRx)$
    \vspace{0.20cm}
    \item [antisymmetric:] $(\forall x,y \in A) (xRyRx \to x=y)$
    \vspace{0.20cm}
    \item [transitive:] $(\forall x,y,z \in A)(xRyRz \to xRz)$
  \end{description}
}

\frame
{
  \frametitle{Equivalence Class and Partition}
  \begin{columns}

  \column{0.50\textwidth}
    \textbf{Equivalence class holds some interesting properties:} \\
    \begin{enumerate}
      \item $\lbrack x \rbrack \neq {\O}$
      \vspace{0.20cm}
      \item $xRy \to \lbrack x \rbrack = \lbrack y \rbrack$
      \vspace{0.20cm}
      \item $\neg (xRy) \to \lbrack x \rbrack \cap \lbrack y \rbrack = {\O}$
      \vspace{0.20cm}
      \item $\cup \lbrace \lbrack x \rbrack \mid x \in A \rbrace = A$
    \end{enumerate}
  \column{0.50\textwidth}
    \textbf{Partition: $\pi \subseteq P(A)$}  \\
    \begin{enumerate}
      \item ${\O} \notin \pi$
      \vspace{0.20cm}
      \item $\forall x \forall y(x,y \in \pi \land x \neq y \to x \cap y = {\O})$
      \vspace{0.20cm}
      \item $\cap \pi = A$
    \end{enumerate}
  \end{columns}
%
%  \pause
  \vspace{0.30cm}
  \begin{quote}
    \textit{There is an one-to-one correspondence between equivalence relation and partition.}
  \end{quote}

  \textbf{7: 如何定义三元组(Optional)?} \\
  \vspace{0.20cm}
  \textbf{We have:}
  \[
    \langle x,y \rangle = \lbrace \lbrace x \rbrace,\lbrace x,y \rbrace \rbrace
  \]
  \textbf{then:}
  \[
    \langle x,y,z \rangle = \lbrace \lbrace x \rbrace, \lbrace x,y \rbrace, \lbrace x,y,z \rbrace \rbrace
  \]
  \pause
  \vspace{0.40cm}

  \textit{Tips: Consider $\langle x,y,x \rangle and \langle x,y,y \rangle$}  \\
  \pause
  \vspace{0.20cm}
  \textbf{A: $\langle x_1,x_2,x_3 \rangle = \langle \langle x_1,x_2 \rangle ,x_3 \rangle.$}
}

  \vspace{0.20cm}

    \vspace{0.30cm}

  \end{enumerate}

  \textbf{9: 请证明如下命题: \scriptsize ($P_{118}$表7.2)}  \\
  \vspace{0.20cm}
  \normalsize
  \[
    R,S \textrm{ are symmetric, so are } R^{-1},R \cap S, R\cup S, \textrm{ and }R-S.

    \item $(R_1 \cap R_2)^{-1} = R_1^{-1} \cap R_2^{-1}.$
  \end{itemize}

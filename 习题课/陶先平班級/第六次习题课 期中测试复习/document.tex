\subsubsection{Closure}
\frame
{
  \frametitle{Closure}

    \textbf{复合闭包:}  \\
    \vspace{0.20cm}
    证明: $A \textrm{ is finite set,} R \subseteq A \times A \Rightarrow rt(R) = tr(R).$  \\
    \vspace{0.20cm}
    \textbf{提示:}  \\
    $(I_A \cup R)^n = I_A \cup R \cup R^2 \cup \cdots \cup R^n = I_A \cup (\cup_{i=1}^{n}R^i)$  \\
    \vspace{0.20cm}
    \textbf{解答:}  \\

  \normalsize
    \begin{equation*}
        \begin{array}{rcl}
            tr(R) & = & t(I_A \cup R)  \\
                  & = & \cup_{i=1}^{n}(I_A \cup R)^{i}  \\
                  & = & (I_A \cup R) \cup (I_A \cup R)^2 \cup \cdots \cup (I_A \cup R)^n  \\
                  & = & (I_A \cup R) \cup (I_A \cup (\cup_{i=1}^{2}R^i)) \cup \cdots \cup (I_A \cup (\cup_{i=1}^{n}R^i))  \\
                  & = & I_A \cup R \cup R^2 \cup \cdots \cup R^n  \\
                  & = & I_A \cup (\cup_{i=1}^{n}R^i)  \\
                  & = & I_A \cup t(R)  \\
                  & = & r(t(R))  \\
                  & = & rt(R)
        \end{array}
    \end{equation*}
}

\frame
{
  \frametitle{Closure}

    \textbf{复合闭包:}  \\
    \vspace{0.20cm}
    \textbf{证明}: $R \subseteq A \times A \Rightarrow st(R) \subseteq ts(R).$  \\
    \vspace{0.40cm}
    \textbf{解答:}  \\

%  \normalsize
    \begin{equation*}
        \begin{array}{rcl}
            R\subseteq s(R)
                  & \Rightarrow & t(R) \subseteq t(s(R))  \\
                  & \Rightarrow & st(R) \subseteq \textcolor{red}{sts(R) = ts(R)}
        \end{array}
    \end{equation*}

    \vspace{0.30cm}
    $s(R)[sym] \Rightarrow ts(R)[sym] \Rightarrow sts(R) = ts(R).$
}

  \textbf{试证明:}  \\
  设$m$是大于1的正整数,记 $U(m) = \lbrace a \in \mathbf{Z}_m \mid (a,m) = 1 \rbrace$,  \\

  \vspace{0.20cm}
  \textbf{举例:}  \\
  \textcolor{gray}{
  \[
    U(15) = \lbrace 1,2,4,7,8,11,13,14 \rbrace
  \]}

  \vspace{0.20cm}

  在求元素$a \in U(m)$的逆元时，你可能会用到如下数论知识:  \\
  \textcolor{blue}{$(a,m) = 1 \Leftrightarrow (\exists u,v \in \mathbf{Z})(au + mv = 1).$} 请说明,$u$即是$a$的逆元.  \\
  \vspace{0.20cm}
  \textbf{解答:}  \\

  \begin{itemize}
    \item \textcolor{red}{运算封闭性}
    \item 结合律
    \item 单位元($1 \in U(m)$)
    \item \textcolor{red}{逆元($au+mv=1, a^{-1} = u.$)}
  \end{itemize}
\end{frame}

        \item \textcolor{gray}{复数乘法的几何意义.}
        \item \textcolor{blue}{先说明$U_n$构成群.}
        \item 令$\omega = \cos \frac {2\pi}{n} + i \sin \frac {2 \pi}{n}$,  \\
          则$U_n = \langle \omega \rangle = \lbrace 1, \omega, \omega^2, \cdots, \omega^{n-1} \rbrace.$
        \item $(k,n) = 1, \omega^k$为生成元.
      \end{enumerate}
  \end{columns}
\end{frame}


\begin{frame}{Cyclic Grooup}
  设$f$为群$(G, \ast)$到群$(H, \circ)$的满同态,  \\
  证明: 若$G$为循环群，则$H$亦为循环群 {\tiny ($P_{204}(27)$}。

  \vspace{0.20cm}
  \textbf{解答:}  \\

  令$G = \langle a \rangle$,则
  \[
    H = f(G) = f(\langle a \rangle) = \lbrace f(a^n) \mid n \in Z \rbrace = \lbrace (f(a))^n \mid n \in Z \rbrace =
    \langle f(a) \rangle.
  \]
\end{frame}



\subsection{Permutation Group}
\begin{frame}{Permutation Group}
  \textbf{试证明:}  \\
  { \scriptsize
  \begin{displaymath}
    \tau = \left( \begin{array}{cccc}
        1 	& 2 	& \cdots & n  \\
        k_1 & k_2 	& \cdots & k_n
      \end{array}\right).
  \end{displaymath}
  则对任一$n$阶置换$\sigma$,有
  \begin{displaymath}
    \sigma^{-1} \tau \sigma = \left( \begin{array}{cccc}
        \sigma(1) 	& \sigma(2) 	& \cdots & \sigma(n)  \\
        \sigma(k_1) & \sigma(k_2) 	& \cdots & \sigma(k_n)
      \end{array}\right).
  \end{displaymath}
  }

  \textbf{解答:}
  置换是函数。如何证明函数相等?  \\

  \vspace{0.30cm}

  \textcolor{blue}{Q: What if $\tau = (a_1, a_2, \cdots, a_k)$?}
\end{frame}

\begin{frame}{Permuation Group}

  { \scriptsize
    $(i_k a \cdots b), (i_n c \cdots d) \textrm{不相交},$则
  \[
    (i_k ,i_n)(i_k, a ,\cdots ,b)(i_n ,c ,\cdots ,d) = (i_k ,a, \cdots ,b ,i_n ,c ,\cdots ,d).
  \]
  \[
    (i_k ,i_n)(i_k ,a ,\cdots, b, i_n, c ,\cdots, d) = (i_k, a ,\cdots, b)(i_n, c ,\cdots ,d).
  \]
  }

  \vspace{0.30cm}

  \emph{简单介绍另一种``置换可表为不相交轮换之积''的证明方法。}
\end{frame}

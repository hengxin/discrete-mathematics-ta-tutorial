\begin{problem}{\textbf{容斥原理}}

    给定集合$\set{1,2,\cdots,2n}(n \in N^+)$($N^+$表示正整数集合)，如果它的某个排列$\pi = \set{x_1,x_2,\cdots,x_{2n}}$
    满足条件
    \[
      \exists i \in \set{1,2,\cdots,2n-1}. (\vert x_i - x_{i+1} \vert = n),
    \]
    则称该排列具有性质$P$.

    \begin{enumerate}
      \item 试使用容斥原理给出具有性质$P$的排列的个数(\emph(可以不给出最简式)).
      \item 若已知有关容斥原理的命题\ref{in-ex}成立，请证明对于任意$n$，具有性质$P$的排列的个数比不具有性质$P$的排列的个数要多。

        \begin{align}  \label{in-ex}
          \textrm{对于有限集合 }S_1,S_2,\cdots,S_n, \vert \cup_{1 \leq i \leq n} S_i \vert \geq \Sigma_{1 \leq i \leq n} \vert
          S_i \vert - \Sigma_{1 \leq i < j \leq n} \vert S_i \cap S_j \vert.
        \end{align}
    \end{enumerate}
\end{problem}
